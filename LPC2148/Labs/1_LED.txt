#include <lpc214x.h>

void init(){
      IO1DIR |= 1 << 16;
      IO0DIR &= ~(1 << 19);
      PINSEL2 &= ~(1 << 3);
      PINSEL0 &= ~(1 << 0);
}

void delay(){
      volatile unsigned int i = 0;
      for (i = 0; i < 1000000; i++);
}

void led_on(){
      IO1SET = 1 << 16;
      delay();
}

void led_off(){
      IO1CLR = 1 << 16;
      delay();
}

void toggle_led(){
      IO1SET = 1 << 16;
      delay();
      IO1CLR = 1 << 16;
      delay();
}

void push_button(){
      static int ok = 0;
      if((IOPIN0 & (1 << 0)) == 0){
            ok = 1 - ok;
            
            if(ok == 1)
                  led_on();
            else
                  led_off();
      }
}

int main(void)
{
      init();
      
      while(1)
      {
            push_button();
      }
      return 0;
}
